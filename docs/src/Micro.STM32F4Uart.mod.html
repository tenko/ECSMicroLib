        UCLK*: INTEGER;
        baud*: INTEGER;

        stopBits*: INTEGER;

Bus* = RECORD (BusUart.Bus)

        enableTCI, disableTCI: SET;
        SR, DR, CR1: ADDRESS;
    END;
